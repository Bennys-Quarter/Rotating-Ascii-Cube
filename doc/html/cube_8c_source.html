247  Vec2 B = projection_2d(V1, 50, 30);
248  Vec2 C = projection_2d(V2, 50, 30);

250 
251  Vec2 tmp = {0.0, 0.0};
252  Vec2 M = {-1.0, -1.0};
253 
254  if (A.z > B.z) { tmp = A; A = B; B = tmp; }
255  if (A.z > C.z) { tmp = A; A = C; C = tmp; }
256  if (B.z > C.z) { tmp = B; B = C; C = tmp; }
